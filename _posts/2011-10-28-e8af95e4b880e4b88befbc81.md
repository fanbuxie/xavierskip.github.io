    import urllib2
    
    #输入域名
    weburl = raw_input("input URL:")
    
    #字符串处理成完整的URL格式
    httpurl = 'http://www.%s/'  %(weburl)
    
    #最简单调用urlopen的方式
    content = urllib2.urlopen(httpurl).read()
    
    #显示在下载的URL
    print   '正在下载URL：'+httpurl+'的页面'
    
    #储存的文件名
    urlname =  weburl+".html"
    #创建并打开文件
    testFile = open(urlname,'a')
    #写入内容
    testFile.write(content)
    #关闭
    testFile.close()
